## Usage Examples

### Example 1: Find Critical Security Findings

```dataviewjs
const criticalFindings = dv.pages()
    .where(p => p.type === "audit" || p.type === "assessment")
    .where(p => p.tags && p.tags.includes("security"))
    .where(p => {
        if (!p.findings) return false;
        return p.findings.some(f =>
            f.severity === "critical" &&
            f.status !== "resolved"
        );
    });

dv.table(["Audit", "Critical Findings"],
    criticalFindings.map(p => [p.file.link, p.findings.length])
);
```

### Example 2: Architecture Health Dashboard

```dataviewjs
const archDocs = dv.pages()
    .where(p => p.category === "architecture")
    .where(p => p.status === "active");

const metrics = {
    total: archDocs.length,
    withTests: archDocs.where(p => p.test_coverage > 0).length,
    recent: archDocs.where(p => {
        const age = (Date.now() - new Date(p.updated_date)) / (1000*60*60*24);
        return age <= 90;
    }).length
};

const healthScore = Math.round(
    ((metrics.withTests / metrics.total) * 50) +
    ((metrics.recent / metrics.total) * 50)
);

dv.header(3, `Architecture Health: ${healthScore}%`);
```
